-- Confident_Beginner_Intermediate_Queries#

-- Quick review of the basics

SELECT 
  GivenName,
  Surname,
  Age,
  CountryFull
FROM "Customers"
WHERE CountryFull = 'Canada'
AND Age > 40
ORDER BY Age DESC;

SELECT 
  CountryFull,
  COUNT(*) AS total_customers
FROM "Customers"
GROUP BY CountryFull
ORDER BY total_customers DESC;

SELECT 
  CountryFull,
  COUNT(*) AS total_customers
FROM "Customers"
GROUP BY CountryFull
HAVING COUNT(*) > 5
ORDER BY total_customers DESC;


-- Applying useful functions

SELECT ROUND(AVG("Age"), 1) AS average_age_rounded
FROM "Customers";

SELECT 
  Gender,
  COUNT(*) AS total_customers,
  ROUND(AVG(Age), 1) AS average_age
FROM "Customers"
GROUP BY Gender;


-- Percentages

SELECT 
  "Gender",
  COUNT(*) AS total,
  ROUND(
    COUNT(*) * 100.0 / SUM(COUNT(*)) OVER (), 
    2
  ) AS percentage
FROM "Customers"
GROUP BY "Gender";


-- String functions
SELECT 
  CONCAT("GivenName", ' ', "Surname") AS full_name
FROM "Customers";

SELECT 
  "GivenName" || ' ' || "Surname" AS full_name,
  "Occupation"
FROM "Customers"
LIMIT 10;

SELECT 
  UPPER("CountryFull") AS country_upper
FROM "Customers";

SELECT 
  LOWER("CountryFull") AS country_upper
FROM "Customers";

-- Good for data cleaning
SELECT 
  TRIM("GivenName") AS cleaned_name
FROM "Customers";

SELECT 
  "GivenName",
  LENGTH("GivenName") AS name_length
FROM "Customers";

-- DATE functions

-- Ensure that your date column is the date data type 
SELECT 
  EXTRACT(YEAR FROM "StartDT") AS signup_year
FROM "Customers";


SELECT 
  EXTRACT(MONTH FROM "StartDT") AS signup_month
FROM "Customers";

SELECT 
  EXTRACT(DOW FROM "StartDT") AS signup_dow
FROM "Customers";

-- Days from sign-up
SELECT
  "GivenName",
  "Surname",
  CURRENT_DATE - "StartDT" AS days_since_signup
FROM "Customers";

SELECT
  "GivenName",
  "Surname",
  CURRENT_DATE - "StartDT" AS days_since_signup
FROM "Customers"
ORDER BY days_since_signup ASC;

-- Current Customers
SELECT
  "GivenName",
  "Surname",
  "StartDT",
  "EndDT",
  CURRENT_DATE - "StartDT" AS days_since_signup
FROM "Customers"
WHERE "EndDT" > CURRENT_DATE;

-- Recent customers
SELECT *
FROM "Customers"
WHERE "StartDT" >= CURRENT_DATE - INTERVAL '3065 days';


-- Current longest membership customers
SELECT
  "GivenName",
  "Surname",
  "StartDT",
  "EndDT",
  CURRENT_DATE - "StartDT" AS days_since_signup
FROM "Customers"
WHERE "EndDT" > CURRENT_DATE
ORDER BY days_since_signup DESC;

-- Make a date using MAKE_DATE and year, month, day components 
-- ::int casts into an integer

SELECT 
  "GivenName",
  "Surname",
  "Birthday",
  MAKE_DATE(
    EXTRACT(YEAR FROM CURRENT_DATE)::int,
    EXTRACT(MONTH FROM "Birthday")::int,
    EXTRACT(DAY FROM "Birthday")::int
  ) AS birthday_this_year
FROM "Customers";

-- Truncating Dates for bucketing into months/years

SELECT 
  DATE_TRUNC('month', "StartDT") AS signup_month,
  COUNT(*) AS total
FROM "Customers"
GROUP BY signup_month
ORDER BY signup_month;

-- CASE / WHEN / END Syntax for sorting

SELECT
  "GivenName",
  "Surname",
  "GivenName",
  "Gender",
  CASE
    WHEN "Gender" ILIKE 'Female' THEN 'Female Member'
    WHEN "Gender" ILIKE 'Male' THEN 'Male Member'
    ELSE 'No Gender Specified Member'
  END AS gender_category
FROM "Customers";

SELECT
"GivenName",
  "Surname",
  "GivenName",
  "Age",
  CASE
    WHEN "Age" < 30 THEN 'Youth Member'
    WHEN "Age" BETWEEN 30 AND 50 THEN 'Adult Member'
    ELSE 'Senior Member'
  END AS age_category
FROM "Customers";

SELECT
  "GivenName",
  "Surname",
  "Age",
  CASE
    WHEN "Age" < 25 THEN 10
    WHEN "Age" BETWEEN 25 AND 60 THEN 5
    ELSE 15
  END AS discount_percentage
FROM "Customers";

SELECT
  "GivenName",
  "Surname",
  "Occupation",
  CASE
    WHEN "Occupation" IS NULL THEN 'Incomplete Record'
    ELSE 'Complete Record'
  END AS data_status
FROM "Customers";

SELECT
  "GivenName",
  "Surname",
  "CountryFull",
  CASE
    WHEN "CountryFull" IN ('United States', 'Canada', 'Australia')
      THEN 'Tier 1 Market'
    ELSE 'Other Market'
  END AS market_category
FROM "Customers";

SELECT
  "CountryFull",
  COUNT(*) FILTER (WHERE "Gender" ILIKE 'Female') AS female_customers,
  COUNT(*) FILTER (WHERE "Gender" ILIKE 'Male') AS male_customers
FROM "Customers"
GROUP BY "CountryFull"
ORDER BY "CountryFull";

SELECT
  "CountryFull",
  COUNT(*) FILTER (WHERE "Age" < 30) AS under_30,
  COUNT(*) FILTER (WHERE "Age" BETWEEN 30 AND 50) AS age_30_50,
  COUNT(*) FILTER (WHERE "Age" > 50) AS over_50
FROM "Customers"
GROUP BY "CountryFull"
ORDER BY "CountryFull";

SELECT
  "CountryFull",
  COUNT(*) AS total,
  COUNT(*) FILTER (WHERE "Gender" ILIKE 'Female') AS female_count,
  ROUND(
    COUNT(*) FILTER (WHERE "Gender" ILIKE 'Female') * 100.0 
    / COUNT(*),
    1
  ) AS female_percentage
FROM "Customers"
GROUP BY "CountryFull";

-- Joins

SELECT
  c."GivenName", -- from customers table
  c."Surname", -- from customers table
  s."OrderDate", -- from sales table
  s."Quantity" -- from sales table
FROM "Sales" s -- note the alias 's'
INNER JOIN "Customers" c -- note the alias 'c'
  ON s."CustomerKey" = c."CustomerKey" -- the keys that create the relationship
ORDER BY s."OrderDate" DESC;


SELECT
  p."ProductName",
  p."Price",
  s."Quantity",
  s."OrderDate"
FROM "Sales" s
INNER JOIN "Products" p
  ON s."ProductKey" = p."ProductKey"
ORDER BY s."OrderDate" DESC;


-- Three-table join
SELECT
  c."GivenName", -- Customers table
  c."Surname",
  p."ProductName", -- products table
  p."Price",
  s."Quantity", -- Sales table
  s."OrderDate"
FROM "Sales" s
INNER JOIN "Customers" c
  ON s."CustomerKey" = c."CustomerKey"
INNER JOIN "Products" p
  ON s."ProductKey" = p."ProductKey"
ORDER BY s."OrderDate" DESC;

-- with item * quantity calculation
SELECT
  s."OrderKey",
  c."GivenName", -- Customers table
  c."Surname",
  p."ProductName", -- products table
  p."Price",
  s."Quantity", -- Sales table
  s."OrderDate",
  p."Price" * s."Quantity" AS item_subtotal
FROM "Sales" s
INNER JOIN "Customers" c
  ON s."CustomerKey" = c."CustomerKey"
INNER JOIN "Products" p
  ON s."ProductKey" = p."ProductKey"
ORDER BY s."OrderDate" DESC;

-- Subquery using aggregation
SELECT 
"OrderKey",
SUM(item_subtotal)
FROM
  (SELECT
    s."OrderKey",
    c."GivenName", -- Customers table
    c."Surname",
    p."ProductName", -- products table
    p."Price",
    s."Quantity", -- Sales table
    s."OrderDate",
    p."Price" * s."Quantity" AS item_subtotal
  FROM "Sales" s
  INNER JOIN "Customers" c
    ON s."CustomerKey" = c."CustomerKey"
  INNER JOIN "Products" p
    ON s."ProductKey" = p."ProductKey")
GROUP BY "OrderKey";

-- Which customers haven’t purchased anything?
SELECT
  c."GivenName",
  c."Surname",
  s."OrderDate"
FROM "Customers" c
LEFT JOIN "Sales" s
  ON c."CustomerKey" = s."CustomerKey"
WHERE s."CustomerKey" IS NULL;

-- same query with null order date
SELECT
  c."GivenName",
  c."Surname",
  s."OrderDate"
FROM "Customers" c
LEFT JOIN "Sales" s
  ON c."CustomerKey" = s."CustomerKey"
WHERE s."OrderDate" IS NULL;

SELECT
  c."GivenName",
  c."Surname",
  SUM(s."Quantity" * s."UnitPrice") AS total_spent
FROM "Sales" s
INNER JOIN "Customers" c
  ON s."CustomerKey" = c."CustomerKey"
INNER JOIN "Products" p
  ON s."ProductKey" = p."ProductKey"
GROUP BY c."GivenName", c."Surname"
ORDER BY total_spent DESC;

-- Most popular products
SELECT
  p."ProductName",
  SUM(s."Quantity") AS total_quantity_sold
FROM "Sales" s
INNER JOIN "Products" p
  ON s."ProductKey" = p."ProductKey"
GROUP BY p."ProductName"
ORDER BY total_quantity_sold DESC;
